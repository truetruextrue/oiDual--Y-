- Gera mesa comparativa direto/espelho

import sys, os, re, json, argparse
from datetime import datetime

def read(path):
    if path and os.path.isfile(path):
        return open(path, "r", encoding="utf-8", errors="ignore").read()
    return ""

def counts(txt):
    lines = txt.splitlines()
    n_lines = len(lines)
    n_chars = len(txt)
    n_letters = len(re.findall(r"[^\W\d_]", txt, flags=re.UNICODE))
    n_digits = len(re.findall(r"\d", txt))
    n_space  = len(re.findall(r"\s", txt))
    n_punct  = len(re.findall(r"[^\w\s]", txt, flags=re.UNICODE))
    words = re.findall(r"\b\w+\b", txt, flags=re.UNICODE)
    n_words = len(words)
    avg_chars_per_line = (n_chars / n_lines) if n_lines else 0.0
    avg_words_per_line = (n_words / n_lines) if n_lines else 0.0
    avg_chars_per_word = (sum(len(w) for w in words)/n_words) if n_words else 0.0
    return {
        "lines": n_lines,
        "chars": n_chars,
        "letters": n_letters,
        "digits": n_digits,
        "spaces": n_space,
        "punct": n_punct,
        "words": n_words,
        "avg_chars_per_line": round(avg_chars_per_line, 3),
        "avg_words_per_line": round(avg_words_per_line, 3),
        "avg_chars_per_word": round(avg_chars_per_word, 3),
    }

def mk_table(direct, mirror):
    rows = [
        ("linhas", direct["lines"], mirror["lines"]),
        ("caracteres", direct["chars"], mirror["chars"]),
        ("letras", direct["letters"], mirror["letters"]),
        ("dígitos", direct["digits"], mirror["digits"]),
        ("espaços", direct["spaces"], mirror["spaces"]),
        ("pontuação", direct["punct"], mirror["punct"]),
        ("palavras", direct["words"], mirror["words"]),
        ("média chars/linha", direct["avg_chars_per_line"], mirror["avg_chars_per_line"]),
        ("média palavras/linha", direct["avg_words_per_line"], mirror["avg_words_per_line"]),
        ("média chars/palavra", direct["avg_chars_per_word"], mirror["avg_chars_per_word"]),
    ]
    out = ["| métrica | direto | espelho |","|---|---:|---:|"]
    out += [f"| {k} | {a} | {b} |" for (k,a,b) in rows]
    return "\n".join(out)

def main():

- Gera n-gramas (bigramas)
- Selo: ∆⁷

import sys, os, re, json, argparse
from datetime import datetime
from collections import Counter

LATIN_RANGE = (0x0041, 0x024F)
CYR_RANGE   = (0x0400, 0x04FF)
GRC_RANGE   = (0x0370, 0x03FF)

STOP_PT = {"de","do","da","que","em","um","uma","para","no","na","os","as","é"}
STOP_EN = {"the","of","to","and","in","a","is","that","for","on","with"}
STOP_ES = {"de","la","que","el","en","y","a","los","se","del","las","un","una"}

DIAC_PT = "áàâãéêíóôõúç"
DIAC_ES = "áéíóúñü"

def read(path):
    if path and os.path.isfile(path):
        return open(path, "r", encoding="utf-8", errors="ignore").read()
    return ""

def guess_script(txt):
    codepoints = [ord(ch) for ch in txt if ch.strip()]
    if any(CYR_RANGE[0] <= cp <= CYR_RANGE[1] for cp in codepoints):
        return "Cyrillic"
    if any(GRC_RANGE[0] <= cp <= GRC_RANGE[1] for cp in codepoints):
        return "Greek"
    return "Latin"

def tokenize(txt):
    return [w.lower() for w in re.findall(r"\b\w+\b", txt, flags=re.UNICODE)]

def score_lang(tokens):
    tset = set(tokens)
    sc_pt = len(tset & STOP_PT) + sum(ch in DIAC_PT for ch in "".join(tokens))
    sc_es = len(tset & STOP_ES) + sum(ch in DIAC_ES for ch in "".join(tokens))
    sc_en = len(tset & STOP_EN)
    scores = {"pt": sc_pt, "es": sc_es, "en": sc_en}
    lang = max(scores, key=scores.get) if any(scores.values()) else "und"
    return lang, scores

def bigrams(tokens, top=10):
    grams = ["".join(pair) for pair in zip(tokens, tokens[1:])]
    c = Counter(grams).most_common(top)
    return [{"ngram": g, "count": n} for g, n in c]

"""

def read_text(path):

def seed_from_text(*chunks):
    h = hashlib.sha256()
    for c in chunks:
        if c: h.update(c.encode("utf-8", errors="ignore"))
    return int(h.hexdigest(), 16) & ((1<<63)-1)

def gen_trinary_grid(size, rng):
    if size % 2 == 0: size += 1
    half = size // 2
    grid = [[0]*size for _ in range(size)]
    for r in range(size):
        row_half = [rng.randrange(0,3) for _ in range(half)]
        center = rng.randrange(0,3)
        row = row_half + [center] + row_half[::-1]
        grid[r] = row
    for r in range(half):
        grid[size-1-r] = list(grid[r])
    return grid

def to_pbm_p1(grid):
    h = len(grid); w = len(grid[0]) if h else 0
    lines = [f"P1", f"{w} {h}"]
    for r in grid:
        row = ["1" if v==1 else "0" for v in r]
        lines.append(" ".join(row))
    return "\n".join(lines)

def to_tri_txt(grid):
    return "\n".join("".join(str(v) for v in row) for row in grid)

def to_ascii_block(grid):
    char = {0:" ", 1:"█", 2:"▒"}
    return "\n".join("".join(char[v] for v in row) for row in grid)

def analyze_invariants(grid):
    size = len(grid)
    if size == 0: return {"pal_h": False, "pal_v": False, "rot_180": False}

    pal_v = all(grid[r][c] == grid[size-1-r][c] for r in range(size) for c in range(size))
    match = 0; total = size*size
    for r in range(size):
        for c in range(size):
            if grid[r][c] == grid[size-1-r][size-1-c]:
                match += 1
    rot180_ratio = round(match/total, 3)
    return {"pal_h": pal_h, "pal_v": pal_v, "rot_180_ratio": rot180_ratio}
