    try:

def mirror_line(s):  # espelha caracteres por linha

    return s[::-1]


def reverse_by_lines(txt):

    return "\n".join(mirror_line(line) for line in txt.splitlines())


def count_basic(txt):

    letters = len(re.findall(r"\S", txt, flags=re.UNICODE))

    words = len(re.findall(r"\b\w+\b", txt, flags=re.UNICODE))

    # frases: ., !, ?, ou linhas vazias como separadores suaves

    sentences = len(re.findall(r"[.!?]+(?=\s|$)", txt)) or txt.count("\n\n")

    return {"letters": letters, "words": words, "sentences": sentences}

        found["D"] = True

    # Coluna 1 (ex.: 11=2+9×1 etc.)

    if re.search(r"\b11\s*=\s*2\s*\+\s*9\s*×\s*1\b", txt):

        found["coluna_1"] = True

    return found

import sys, os, re, json, argparse

from collections import Counter


LATIN_RANGE = (0x0041, 0x024F)

CYR_RANGE   = (0x0400, 0x04FF)

GRC_RANGE   = (0x0370, 0x03FF)


STOP_PT = {"de","do","da","que","em","um","uma","para","no","na","os","as","é"}

STOP_EN = {"the","of","to","and","in","a","is","that","for","on","with"}

STOP_ES = {"de","la","que","el","en","y","a","los","se","del","las","un","una"}


DIAC_PT = "áàâãéêíóôõúç"

DIAC_ES = "áéíóúñü"


def read(path):

        return open(path, "r", encoding="utf-8", errors="ignore").read()

    return ""


def guess_script(txt):

    codepoints = [ord(ch) for ch in txt if ch.strip()]

    if any(CYR_RANGE[0] <= cp <= CYR_RANGE[1] for cp in codepoints): return "Cyrillic"

    if any(GRC_RANGE[0] <= cp <= GRC_RANGE[1] for cp in codepoints): return "Greek"

    return "Latin"


def tokenize(txt):

    return [w.lower() for w in re.findall(r"\b\w+\b", txt, flags=re.UNICODE)]


def score_lang(tokens):

    tset = set(tokens)

    sc_pt = len(tset & STOP_PT) + sum(ch in DIAC_PT for ch in "".join(tokens))

    sc_es = len(tset & STOP_ES) + sum(ch in DIAC_ES for ch in "".join(tokens))

    sc_en = len(tset & STOP_EN)

    scores = {"pt": sc_pt, "es": sc_es, "en": sc_en}

    lang = max(scores, key=scores.get) if any(scores.values()) else "und"

    return lang, scores


def bigrams(tokens, top=10):

    grams = ["".join(pair) for pair in zip(tokens, tokens[1:])]

    c = Counter(grams).most_common(top)

    return [{"ngram": g, "count": n} for g,n in c]

def repunit(n, b=10):

    if b <= 1 or n <= 0: return 0

    return (b**n - 1) // (b - 1)


def is_palindrome(s):

def seed_from_text(*chunks):

    h = hashlib.sha256()

    for c in chunks:

        if c: h.update(c.encode("utf-8", errors="ignore"))

    return int(h.hexdigest(), 16) & ((1<<63)-1)


def gen_trinary_grid(size, rng):

    """Gera matriz size×size (size ímpar), palindrômica H/V e approx. rotacional.

       Valores em {0,1,2}. Constrói metade e espelha."""

    if size % 2 == 0: size += 1

    half = size // 2

    grid = [[0]*size for _ in range(size)]

    # primeira meia-linha aleatória e espelha

    for r in range(size):

        row_half = [rng.randrange(0,3) for _ in range(half)]

        center = rng.randrange(0,3)

        row = row_half + [center] + row_half[::-1]

        grid[r] = row

    # agora espelha verticalmente para garantir H/V palíndromo global

    for r in range(half):

        grid[size-1-r] = list(grid[r])

    # opcional: pequeno "twist" rotacional (mesmo valor no diametral)

    for r in range(size):

        for c in range(size):

4) Como “interpretar” em binário/trinário


PBM P1: primeira linha P1, segunda width height, depois linhas com 0 e 1.

Qualquer parser Netpbm ou script simples lê e reconstrói a imagem binária.


TRI TXT: cada linha é uma sequência de 0|1|2. Pode mapear:


0→silêncio, 1→pulso, 2→sustentado,


ou 0→baixo, 1→médio, 2→alto (para TTS/sons).



SCI TXT: cabeçalho com key=value e bloco ASCII palindrômico (o mesmo “pulso” lido esquerda→direita e direita→esquerda).
